 
#include <stdio.h>

void soinsuu(int n);

int main (void){
  int n;
  printf("n> ");
  scanf("%d", &n);
  soinsuu(n);
  return 0;
}

void soinsuu(int n) {
  int i = 2;
  while(n >= i) {
    if(n % i != 0) {
      i++;
    } else {
      n /= i;
      printf("%d ", i);
    }
  }
  printf("\n");
}
